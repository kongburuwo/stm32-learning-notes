# 初始化时钟

HAL_RCC_OscConfig();   // 负责：造水（产生时钟）
HAL_RCC_ClockConfig(); // 负责：送水（使用时钟）


*主要函数*
HAL_StatusTypeDef HAL_RCC_OscConfig(RCC_OscInitTypeDef  *RCC_OscInitStruct)

HAL_StatusTypeDef HAL_RCC_ClockConfig(RCC_ClkInitTypeDef  *RCC_ClkInitStruct, uint32_t FLatency)


*相关c文件*
```c
#include "sys.h"

//开启HSE → 配置PLL → 配置SYSCLK/总线分频 → 配置Flash等待周期
void stm32_clock_init(uint32_t pllmul)
{
	
	
	HAL_StatusTypeDef ret = HAL_ERROR;
	
	RCC_OscInitTypeDef rcc_osc_init = {0};
	RCC_ClkInitTypeDef rcc_clk_init = {0};
	
	// 打开外部8MHz晶振
	__HAL_RCC_HSE_CONFIG(RCC_HSE_ON);
	// 死等晶振稳定（不稳定就一直卡在这里）
  while(__HAL_RCC_GET_FLAG(RCC_FLAG_HSERDY) == RESET);
	
	//配置振荡器（HSE + PLL）
	rcc_osc_init.OscillatorType = RCC_OSCILLATORTYPE_HSE;
	rcc_osc_init.HSEState = RCC_HSE_ON;
	rcc_osc_init.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
	rcc_osc_init.PLL.PLLState = RCC_PLL_ON;
	rcc_osc_init.PLL.PLLSource = RCC_PLLSOURCE_HSE;
	rcc_osc_init.PLL.PLLMUL = pllmul;
	
	ret = HAL_RCC_OscConfig(&rcc_osc_init);
	if(ret != HAL_OK)
	{
		while(1);
	}
	
	//配置系统时钟树（SYSCLK、HCLK、PCLK1/2）
	rcc_clk_init.ClockType = RCC_CLOCKTYPE_SYSCLK | RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
	rcc_clk_init.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
	rcc_clk_init.AHBCLKDivider = RCC_SYSCLK_DIV1;
	rcc_clk_init.APB1CLKDivider = RCC_HCLK_DIV2;
	rcc_clk_init.APB2CLKDivider = RCC_HCLK_DIV1;
	
	ret = HAL_RCC_ClockConfig(&rcc_clk_init,FLASH_LATENCY_2);
	
		if(ret != HAL_OK)
	{
		while(1);
	}
	
	
}
```
